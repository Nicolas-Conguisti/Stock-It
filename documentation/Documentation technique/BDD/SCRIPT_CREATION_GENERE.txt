CREATE TABLE UTILISATEUR(
   UTILISATEUR_ID INT,
   UTILISATEUR_NOM VARCHAR(70) NOT NULL,
   UTILISATEUR_PRENOM VARCHAR(70) NOT NULL,
   UTILISATEUR_MAIL VARCHAR(100) NOT NULL,
   UTILISATEUR_MDP VARCHAR(200) NOT NULL,
   PRIMARY KEY(UTILISATEUR_ID)
);

CREATE TABLE PROJET(
   PROJET_ID SMALLINT,
   PROJET_NOM VARCHAR(100) NOT NULL,
   PROJET_DESCRIPTION VARCHAR(200),
   PROJET_ADRESSE VARCHAR(200),
   PROJET_CODEPOSTAL BYTE,
   PROJET_STATUT VARCHAR(50) NOT NULL,
   PROJET_TYPE VARCHAR(50) NOT NULL,
   PROJET_DATEDEB DATE NOT NULL,
   PROJET_DATEFIN DATE,
   UTILISATEUR_ID INT NOT NULL,
   PRIMARY KEY(PROJET_ID),
   FOREIGN KEY(UTILISATEUR_ID) REFERENCES UTILISATEUR(UTILISATEUR_ID)
);

CREATE TABLE LIEUSTOCK(
   LIEUSTOCK_ID BYTE,
   LIEUSTOCK_NOM VARCHAR(70) NOT NULL,
   LIEUSTOCK_ESPACETOTAL SMALLINT,
   LIEUSTOCK_ESPACEDISPO SMALLINT,
   LIEUSTOCK_ADRESSE VARCHAR(200),
   LIEUSTOCK_CODEPOSTAL BYTE,
   PROJET_ID SMALLINT NOT NULL,
   PRIMARY KEY(LIEUSTOCK_ID),
   FOREIGN KEY(PROJET_ID) REFERENCES PROJET(PROJET_ID)
);

CREATE TABLE OBJET(
   OBJET_ID INT,
   OBJET_NOM VARCHAR(100) NOT NULL,
   OBJET_MARQUE VARCHAR(100),
   OBJET_REFERENCE VARCHAR(200),
   OBJET_PRIX DECIMAL(8,2),
   OBJET_QUANTITE VARCHAR(20),
   OBJET_UTILISATION VARCHAR(200),
   OBJET_LONGUEUR BYTE,
   OBJET_LARGEUR BYTE,
   OBJET_HAUTEUR BYTE,
   OBJET_ID_CONTENANT INT,
   LIEUSTOCK_ID BYTE NOT NULL,
   PRIMARY KEY(OBJET_ID),
   FOREIGN KEY(OBJET_ID_CONTENANT) REFERENCES OBJET(OBJET_ID),
   FOREIGN KEY(LIEUSTOCK_ID) REFERENCES LIEUSTOCK(LIEUSTOCK_ID)
);
